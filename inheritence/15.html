abstract class Animal
{
  abstract void sound();
}
class Lion extends Animal
{
   void sound()
    {
       System.out.println("Lion roars");
   }
}
class Tiger extends Animal
{
   void sound()
    {
       System.out.println("Tiger growls");
   }
}
class main
{
  public static void main(String st[])
    {
       Animal obj=new Lion();
        Animal obj1=new Tiger();
               obj.sound();
              obj1.sound();
   }
}

OUTPUT:
Lion roars
Tiger growls
