    public static int findLargest(int arr[])
    {
    Arrays.sort(arr);
    if(arr.length==0)
    {
    System.out.println("Array is empty");
    return -1;
    }
    return arr[arr.length-1];
    }
    
    public static void printArray(int arr[])
    {
    for(int i=0;i<arr.length;i++)
    {
    System.out.print(arr[i] + " ");
    }
    }
