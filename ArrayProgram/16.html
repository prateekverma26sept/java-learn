    public static int findThirdLargest(int arr[])
    {
    
    Arrays.sort(arr);
    if(arr.length<3)
    {
    System.out.println("Array does not have 3 elements");
    return -1;
    }
    return arr[arr.length-3];
    }

    {
    System.out.print(arr[i]+ " ");
    }
