    public static String findDuplicate(String str)
    {
    String words[]=str.split(" ");
    String duplicateWords=" ";
    
    for(int i=0;i<words.length;i++)
    {
    for(int j=i+1;j<words.length;j++)
    {
    if(words[i].equals(words[j]) &&! duplicateWords.contains(words[i]))
    {
    duplicateWords+=words[i] + " ";
    }
    }
    }
    return duplicateWords.trim();
    }
    }
