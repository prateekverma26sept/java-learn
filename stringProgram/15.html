    public static int countPunctuation(String str)
    {
    int count=0;
    for(int i=0;i<str.length();i++)
    {
    char c=str.charAt(i);
    if((c >=32 && c <=47) || (c >=58 && c <=64) || (c >=91 && c <=96) || (c >=123 && c <=126))
    {
    count++;
    }
    }
    return count;
    }
    }
