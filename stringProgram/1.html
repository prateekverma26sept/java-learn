    {
    public static void main(String st[])
    {
    System.out.println(containDigit("123abd"));

    public static boolean containDigit(String t)
    {
    for(int i=0;i < t.length();i++)
    {
    if(!Character.isDigit(t.charAt(i)))
    {
    return false;
    }
    }
    
    return true;
    }
    }
