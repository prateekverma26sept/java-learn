class main21
    {
    public static void main(String st[])
    {
    int originalArray[]={1,2,3,4,5};
    int swappedArray[]=swapFirstAndLast(originalArray);
    
    System.out.println("Original array:" );
    printArray(originalArray);
    
    System.out.println("swapped array:" );
    printArray(swappedArray);
    }
    
    public static int[] swapFirstAndLast(int[] array)
    {
    int[] swappedArray= new int[array.length] ;
    for(int i=0;i<array.length;i++)
    {
    if(i==0)
    {
    swappedArray[i]=array[array.length-1];
    }else if(i==array.length-1)
    {
    swappedArray[i]=array[0];
    }else
    {
    swappedArray[i]=array[i];
    }
    }
    return swappedArray;
    }
    
    public static void printArray(int[] array)
    {
    for(int i=0;i<array.length;i++)
    {
    System.out.print(array[i]+ " ");
    }
    System.out.println();
    }
    }

    OUTPUT:
    Original array:
    1 2 3 4 5
    swapped array:
    5 2 3 4 1
