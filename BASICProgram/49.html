import java.util.Scanner;

    class main49
    {
    public static void main(String st[])
    {
    Scanner obj= new Scanner(System.in);
    System.out.println("Enter sales amount:");
    double sales=obj.nextDouble();
    
    System.out.println("Enter commission rate(in %):");
    double commissionRate=obj.nextDouble();
    
    double commission=calculateCommission(sales,commissionRate);
    System.out.println("Commission:" + commission);
    System.out.println("Commission percentage:" + commissionRate + "%");
    
    
    }
    
    public static double  calculateCommission(double sales, double commissionRate)
    {
    return sales*(commissionRate/100);
    }
    
    }
