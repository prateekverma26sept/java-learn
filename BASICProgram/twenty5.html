    public static boolean isPallindrome(int num)
    {
    int reversedNum=0;
    int og=num;
    
    while(num!=0)
    {
    int digit=num%10;
    reversedNum=reversedNum*10+digit;
    num/=10;
    }
    return og==reversedNum;
    }
    }
