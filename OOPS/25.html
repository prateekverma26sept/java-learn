class ClaExample {

        public static void main(String st[]) {
            
            if (st.length > 0) {
                System.out.println("Command line arguments:");
    
               
                for (int i = 0; i < st.length; i++) {
                    System.out.println("Argument " + (i + 1) + ": " + st[i]);
                }
            } else {
                System.out.println("No command line arguments provided.");
            }
        }
    }
