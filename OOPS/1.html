class Person 
    {
    
    
         String name;
     
        int age;
    
     
        Person(String name, int age)
        {
     
           this.name=name;
            this.age=age ; 
       }
    
        
         
    
        public static void main(String st[])
        {
     
    
            Person P1 = new Person("shanaya", 30);
    
            Person P2 = new Person("anaya", 25);
    
    
            System.out.println("P1: Name=" + P1.name + " ,age:" + P1.age);
    
            System.out.println("P2: Name=" + P2.name + " ,age:" + P2.age);
        }
    
    }

    OUTPUT:
    P1: Name=shanaya ,age=30
    P2: Name=anaya ,age=25
