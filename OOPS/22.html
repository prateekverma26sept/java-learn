class copy
    {
    int x;
    copy(int y)
    {
    x=y;
    }
    copy(copy t)
    {
    x=t.x;
    }
    void display()
    {
    System.out.println(x);
    }
    public static void main(String st[])
    {
    copy obj= new copy(10);
    obj.display();
    obj.x++;
    copy obj1=new copy(obj);
    obj1.display();
    }
    
    
    }

    OUTPUT:
    10
    11
