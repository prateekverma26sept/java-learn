class main8
    {
    public static void main(String st[])
    {
    int arr[][]={
          {1,2,3,4},
          {23,56,21,11},
          {12,89,90,100},
          {45,78,55,13}
    };
    
    int evenCount=0;
    int oddCount=0;
    
    for(int i=0;i<arr.length;i++)
    {
    for(int j=0;j<arr[i].length;j++)
    {
     if(arr[i][j]%2==0)
    {
     evenCount++;
    }else
    {
    oddCount++;
    }
    }
    }
    
    System.out.println("Frequency of even number:"+ evenCount);
    
    System.out.println("Frequency of odd number:"+ oddCount);
    }
    }
