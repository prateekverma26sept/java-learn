#include <stdio.h>
    
    void display(int rows, int cols, int mat[rows][cols]) {
        int i, j;
        for (i = 0; i < rows; i++) {
            for (j = 0; j < cols; j++) {
                printf("%d\t", mat[i][j]);
            }
            printf("\n");
        }
    }

    int main() {
        int rows, cols, i, j;

        // Get the number of rows and columns from the user
        printf("Enter number of rows: ");
        scanf("%d", &rows);
        printf("Enter number of columns: ");
        scanf("%d", &cols);

        int mat[rows][cols];

        // Input matrix elements from the user
        for (i = 0; i < rows; i++) {
            for (j = 0; j < cols; j++) {
                printf("Enter element at Row %d Column %d: ", i + 1, j + 1);
                scanf("%d", &mat[i][j]);
            }
        }

        // Display the matrix
        printf("\nThe Matrix is:\n");
        display(rows, cols, mat);

        return 0;
    }
