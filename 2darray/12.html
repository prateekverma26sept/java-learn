#include <stdio.h>

    void displayTranspose(int rows, int cols, int mat[rows][cols]) {
        int i, j;
        printf("The Transpose of the Matrix is:\n");
        for (j = 0; j < cols; j++) {
            for (i = 0; i < rows; i++) {
                printf("%d\t", mat[i][j]);
            }
            printf("\n");
        }
    }

    int main() {
        int rows, cols, i, j;

        // Get dimensions for the matrix
        printf("Enter number of rows: ");
        scanf("%d", &rows);
        printf("Enter number of columns: ");
        scanf("%d", &cols);

        int mat[rows][cols];

        // Input elements for the matrix
        printf("Enter elements of the matrix:\n");
        for (i = 0; i < rows; i++) {
            for (j = 0; j < cols; j++) {
                printf("Enter element at Row %d Column %d: ", i + 1, j + 1);
                scanf("%d", &mat[i][j]);
            }
        }

        // Display the transpose of the matrix
        displayTranspose(rows, cols, mat);

        return 0;
    }
