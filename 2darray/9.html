#include <stdio.h>

    void calculateRowColumnSums(int rows, int cols, int mat[rows][cols]) {
        int i, j;
        int rowSum, colSum;

        // Calculate sum of each row
        printf("Sum of each row:\n");
        for (i = 0; i < rows; i++) {
            rowSum = 0;
            for (j = 0; j < cols; j++) {
                rowSum += mat[i][j];
            }
            printf("Row %d sum = %d\n", i + 1, rowSum);
        }

        // Calculate sum of each column
        printf("\nSum of each column:\n");
        for (j = 0; j < cols; j++) {
            colSum = 0;
            for (i = 0; i < rows; i++) {
                colSum += mat[i][j];
            }
            printf("Column %d sum = %d\n", j + 1, colSum);
        }
    }

    int main() {
        int rows, cols, i, j;

        // Get dimensions for the matrix
        printf("Enter number of rows: ");
        scanf("%d", &rows);
        printf("Enter number of columns: ");
        scanf("%d", &cols);

        int mat[rows][cols];

        // Input elements for the matrix
        printf("Enter elements of the matrix:\n");
        for (i = 0; i < rows; i++) {
            for (j = 0; j < cols; j++) {
                printf("Enter element at Row %d Column %d: ", i + 1, j + 1);
                scanf("%d", &mat[i][j]);
            }
        }

        // Calculate and display row and column sums
        calculateRowColumnSums(rows, cols, mat);

        return 0;
    }
