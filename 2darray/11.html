#include <stdio.h>
    #include <stdbool.h>

    bool isIdentityMatrix(int size, int mat[size][size]) {
        int i, j;
        for (i = 0; i < size; i++) {
            for (j = 0; j < size; j++) {
                if ((i == j && mat[i][j] != 1) || (i != j && mat[i][j] != 0)) {
                    return false;
                }
            }
        }
        return true;
    }

    int main() {
        int size, i, j;

        // Get dimension for the square matrix
        printf("Enter the size of the square matrix: ");
        scanf("%d", &size);

        int mat[size][size];

        // Input elements for the matrix
        printf("Enter elements of the matrix:\n");
        for (i = 0; i < size; i++) {
            for (j = 0; j < size; j++) {
                printf("Enter element at Row %d Column %d: ", i + 1, j + 1);
                scanf("%d", &mat[i][j]);
            }
        }

        // Check if it is an identity matrix
        if (isIdentityMatrix(size, mat)) {
            printf("\nThe matrix is an Identity Matrix.\n");
        } else {
            printf("\nThe matrix is NOT an Identity Matrix.\n");
        }

        return 0;
    }
