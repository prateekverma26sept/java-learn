#include <stdio.h>

    void display(int rows, int cols, int mat[rows][cols]) {
        int i, j;
        for (i = 0; i < rows; i++) {
            for (j = 0; j < cols; j++) {
                printf("%d\t", mat[i][j]);
            }
            printf("\n");
        }
    }

    int main() {
        int rows1, cols1, rows2, cols2, i, j, k;

        // Get dimensions for the first matrix
        printf("Enter number of rows for the first matrix: ");
        scanf("%d", &rows1);
        printf("Enter number of columns for the first matrix: ");
        scanf("%d", &cols1);

        // Get dimensions for the second matrix
        printf("Enter number of rows for the second matrix: ");
        scanf("%d", &rows2);
        printf("Enter number of columns for the second matrix: ");
        scanf("%d", &cols2);

        // Check if matrix multiplication is possible
        if (cols1 != rows2) {
            printf("Matrix multiplication not possible with these dimensions.\n");
            return 0;
        }

        int mat1[rows1][cols1], mat2[rows2][cols2], product[rows1][cols2];

        // Initialize product matrix to 0
        for (i = 0; i < rows1; i++) {
            for (j = 0; j < cols2; j++) {
                product[i][j] = 0;
            }
        }

        // Input elements for the first matrix
        printf("Enter elements of the first matrix:\n");
        for (i = 0; i < rows1; i++) {
            for (j = 0; j < cols1; j++) {
                printf("Enter element at Row %d Column %d: ", i + 1, j + 1);
                scanf("%d", &mat1[i][j]);
            }
        }

        // Input elements for the second matrix
        printf("Enter elements of the second matrix:\n");
        for (i = 0; i < rows2; i++) {
            for (j = 0; j < cols2; j++) {
                printf("Enter element at Row %d Column %d: ", i + 1, j + 1);
                scanf("%d", &mat2[i][j]);
            }
        }

        // Multiply the matrices
        for (i = 0; i < rows1; i++) {
            for (j = 0; j < cols2; j++) {
                for (k = 0; k < cols1; k++) {
                    product[i][j] += mat1[i][k] * mat2[k][j];
                }
            }
        }

        // Display the product of matrices
        printf("\nThe Product of the matrices is:\n");
        display(rows1, cols2, product);

        return 0;
    }
