#include <stdio.h>

    void displayUpperTriangular(int size, int mat[size][size]) {
        int i, j;
        for (i = 0; i < size; i++) {
            for (j = 0; j < size; j++) {
                if (j >= i) {
                    printf("%d\t", mat[i][j]);
                } else {
                    printf("0\t");
                }
            }
            printf("\n");
        }
    }

    int main() {
        int size, i, j;

        // Get the dimension for the square matrix
        printf("Enter the size of the square matrix: ");
        scanf("%d", &size);

        int mat[size][size];

        // Input elements for the matrix
        printf("Enter elements of the matrix:\n");
        for (i = 0; i < size; i++) {
            for (j = 0; j < size; j++) {
                printf("Enter element at Row %d Column %d: ", i + 1, j + 1);
                scanf("%d", &mat[i][j]);
            }
        }

        // Display the upper triangular matrix
        printf("\nThe Upper Triangular Matrix is:\n");
        displayUpperTriangular(size, mat);

        return 0;
    }
