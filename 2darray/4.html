#include <stdio.h>

    void display(int rows, int cols, int mat[rows][cols]) {
        int i, j;
        for (i = 0; i < rows; i++) {
            for (j = 0; j < cols; j++) {
                printf("%d\t", mat[i][j]);
            }
            printf("\n");
        }
    }

    int main() {
        int rows, cols, i, j;

        // Get dimensions for the matrices
        printf("Enter number of rows: ");
        scanf("%d", &rows);
        printf("Enter number of columns: ");
        scanf("%d", &cols);

        int mat1[rows][cols], mat2[rows][cols], difference[rows][cols];

        // Input elements for the first matrix
        printf("Enter elements of the first matrix:\n");
        for (i = 0; i < rows; i++) {
            for (j = 0; j < cols; j++) {
                printf("Enter element at Row %d Column %d: ", i + 1, j + 1);
                scanf("%d", &mat1[i][j]);
            }
        }

        // Input elements for the second matrix
        printf("Enter elements of the second matrix:\n");
        for (i = 0; i < rows; i++) {
            for (j = 0; j < cols; j++) {
                printf("Enter element at Row %d Column %d: ", i + 1, j + 1);
                scanf("%d", &mat2[i][j]);
            }
        }

        // Subtract the matrices
        for (i = 0; i < rows; i++) {
            for (j = 0; j < cols; j++) {
                difference[i][j] = mat1[i][j] - mat2[i][j];
            }
        }

        // Display the difference of matrices
        printf("\nThe Difference of the matrices is:\n");
        display(rows, cols, difference);

        return 0;
    }
