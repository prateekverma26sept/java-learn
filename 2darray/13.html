#include <stdio.h>

    int isSparseMatrix(int rows, int cols, int mat[rows][cols]) {
        int i, j, zeroCount = 0;
        int totalElements = rows * cols;

        for (i = 0; i < rows; i++) {
            for (j = 0; j < cols; j++) {
                if (mat[i][j] == 0) {
                    zeroCount++;
                }
            }
        }

        // A matrix is sparse if more than half of its elements are zero
        return zeroCount > (totalElements / 2);
    }

    int main() {
        int rows, cols, i, j;

        // Get dimensions for the matrix
        printf("Enter number of rows: ");
        scanf("%d", &rows);
        printf("Enter number of columns: ");
        scanf("%d", &cols);

        int mat[rows][cols];

        // Input elements for the matrix
        printf("Enter elements of the matrix:\n");
        for (i = 0; i < rows; i++) {
            for (j = 0; j < cols; j++) {
                printf("Enter element at Row %d Column %d: ", i + 1, j + 1);
                scanf("%d", &mat[i][j]);
            }
        }

        // Check if the matrix is sparse
        if (isSparseMatrix(rows, cols, mat)) {
            printf("\nThe matrix is a Sparse Matrix.\n");
        } else {
            printf("\nThe matrix is NOT a Sparse Matrix.\n");
        }

        return 0;
    }
