class Average
{
    void CalculateAverage(double a,double b,double c)
      {
           double sum=a+b+c;
           double average=sum/3;
        System.out.println("average of three number:"+average);
       }
}
class main
{
   public static void main(String st[])
     {
             Average obj=new Average();
            obj.CalculateAverage(23.1,1.34,2.1);
      }
}

OUTPUT:
average of three number:8.84666666668
