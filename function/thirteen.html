class main13
{
public static void main(String st[])
{
System.out.println(areVowel("aeiou"));
}

public static boolean areVowel(String t)
{
t=t.toLowerCase();
for(int i=0;i<t.length();i++)
{
char c=t.charAt(i);
if(!(c=='a' || c=='e' || c=='i' || c=='o' || c=='u'))
{
return false;
}
}
return true;
}
}

OUTPUT:
true
