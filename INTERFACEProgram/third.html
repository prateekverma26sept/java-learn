interface Flyable
{
    void fly_obj();
 }
class Spacecraft implements Flyable
{
   public void fly_obj()
     {
         System.out.println("Spacecrfat");
     }
   }
class  Airplane implements Flyable
{
   public void fly_obj()
     {
         System.out.println(" Airplane");
     }
   }

class Helicopter  implements Flyable
{
   public void fly_obj()
     {
         System.out.println("Helicopter ");
     }
   }
class main
{
 public static void main(String st[])
   {
      Flyable obj=new Spacecraft();
      Flyable obj1=new Airplane();
      Flyable obj2=new Helicopter ();
                    obj.fly_obj();
                    obj1.fly_obj();
                    obj2.fly_obj();
  }
}
