interface Playable
    { 
       void play();
    }
   
   class Football implements Playable
     {
        public void play()
           {
              System.out.println("children are playing football");
            }
       }
   class Vollyball implements Playable
     {
         public void play()
           {
              System.out.println("children are playing vollyball");
            }
       }
   class Basketball implements Playable
     {
         public void play()
           {
              System.out.println("children are playing basketball");
            }
       }
    
   class main
   {
     public static void main(String st[])
      {
         Playable obj=new Football();
         Playable obj1=new Vollyball();
         Playable obj2=new Basketball();
                    obj.play();
                    obj1.play();
                    obj2.play();
        }
   }

   
   OUTPUT:
   children are playing football
   children are playing vollyball
   children are playing basketball
